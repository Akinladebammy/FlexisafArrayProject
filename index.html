push()

        

          Mutates Array
        

        
Adds one or more elements to the end of an array and returns the new length.

        
array.push(element);

      


      

        
pop()

        

          Mutates Array
        

        
Removes the last element from an array and returns that element.

        
const lastItem = array.pop();

      


      

        
unshift()

        

          Mutates Array
        

        
Adds one or more elements to the beginning of an array and returns the new length.

        
array.unshift(element);

      


      

        
shift()

        

          Mutates Array
        

        
Removes the first element from an array and returns that element.

        
const firstItem = array.shift();

      


      

        
map()

        

          Returns New Array
        

        
Creates a new array with the results of calling a function on every element.

        
const newArray = array.map(item => item * 2);

      


      

        
filter()

        

          Returns New Array
        

        
Creates a new array with elements that pass the test in a testing function.

        
const filtered = array.filter(item => item > 2);

      


      

        
reduce()

        

          Returns Value
        

        
Applies a function against an accumulator and each element to reduce to a single value.

        
const sum = array.reduce((acc, curr) => acc + curr, 0);

      


      

        
forEach()

        

          Iteration
        

        
Executes a provided function once for each array element.

        
array.forEach(item => console.log(item));

      


      

        
slice()

        

          Returns New Array
        

        
Returns a shallow copy of a portion of an array into a new array.

        
const portion = array.slice(1, 3);

      


      

        
splice()

        

          Mutates Array
        

        
Changes array content by removing/replacing elements and/or adding new elements.

        
array.splice(index, howMany, ...elementsToAdd);
